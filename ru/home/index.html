Пример кода

// Folltawn theme switcher
const themes = ['light', 'dark', 'night'];
let currentTheme = localStorage.getItem('theme') || 'dark';

function cycleTheme() {
    let idx = themes.indexOf(currentTheme);
    idx = (idx + 1) % themes.length;
    setTheme(themes[idx]);
}
